% Examples from The TeXBook, page 154

% \mathcode‘<="313C
% ASCII `<`: 60 (decimal) 3C (hexadecimal)
% class: 3
% family: 1
% character: 60
\directlua{
  print(tex.getmathcodes(60)) -- 3 1 60
}

% \mathcode‘*="2203
% ASCII `*`: 42 (decimal) 2A (hexadecimal)
% class: 2
% family: 2
% character: 3
\directlua{
  print(tex.getmathcodes(42)) -- 2 2 3
}
\bye
